Documentation

Client = require('./lib/client')

var client = new Client(ONENAME_APP_ID, ONENAME_APP_SECRET)

USERS

Lookup users

client.getUser("fredwilson", function(error, response){
    if (error)
        console.log(error)
    else
        console.log(response);
});

Search for users

client.searchUser("wenger", function(error, response){
    if (error)
        console.log(error)
    else
        console.log(response);
});
